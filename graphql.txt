BookController.java
java
package com.example.demo.controller;

import com.example.demo.model.Book;
import org.springframework.graphql.data.method.annotation.Argument;
import org.springframework.graphql.data.method.annotation.MutationMapping;
import org.springframework.graphql.data.method.annotation.QueryMapping;
import org.springframework.stereotype.Controller;

import java.util.ArrayList;
import java.util.List;
import java.util.UUID;

@Controller
public class BookController {

    private final List<Book> books = new ArrayList<>();

    @QueryMapping
    public List<Book> books() {
        return books;
    }

    @QueryMapping
    public Book bookById(@Argument String id) {
        return books.stream()
                .filter(book -> book.getId().equals(id))
                .findFirst()
                .orElse(null);
    }

    @MutationMapping
    public Book addBook(@Argument String title, @Argument String author) {
        Book book = new Book(UUID.randomUUID().toString(), title, author);
        books.add(book);
        return book;
    }
}

5. Service
java
package com.example.graphqlmysql.service;

import com.example.graphqlmysql.entity.Book;
import com.example.graphqlmysql.input.BookInput;
import com.example.graphqlmysql.repository.BookRepository;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
public class BookService {

    private final BookRepository bookRepository;

    public BookService(BookRepository bookRepository) {
        this.bookRepository = bookRepository;
    }

    public List<Book> findAll() {
        return bookRepository.findAll();
    }

    public Book findById(Long id) {
        return bookRepository.findById(id).orElse(null);
    }

    public Book create(BookInput input) {
        Book book = new Book();
        book.setTitle(input.getTitle());
        book.setAuthor(input.getAuthor());
        book.setPrice(input.getPrice());
        return bookRepository.save(book);
    }

    public Book update(Long id, BookInput input) {
        Book book = bookRepository.findById(id)
                .orElseThrow(() -> new RuntimeException("Book tidak ditemukan"));

        book.setTitle(input.getTitle());
        book.setAuthor(input.getAuthor());
        book.setPrice(input.getPrice());

        return bookRepository.save(book);
    }

    public Boolean delete(Long id) {
        if (!bookRepository.existsById(id)) {
            return false;
        }
        bookRepository.deleteById(id);
        return true;
    }
}
6. Controller GraphQL
@QueryMapping adalah shortcut untuk field di type Query, dan @MutationMapping adalah shortcut untuk field di type Mutation. Spring Boot otomatis mendeteksi class @Controller dan mendaftarkan method-method itu sebagai DataFetcher.

java
package com.example.graphqlmysql.controller;

import com.example.graphqlmysql.entity.Book;
import com.example.graphqlmysql.input.BookInput;
import com.example.graphqlmysql.service.BookService;
import jakarta.validation.Valid;
import org.springframework.graphql.data.method.annotation.Argument;
import org.springframework.graphql.data.method.annotation.MutationMapping;
import org.springframework.graphql.data.method.annotation.QueryMapping;
import org.springframework.stereotype.Controller;

import java.util.List;

@Controller
public class BookController {

    private final BookService bookService;

    public BookController(BookService bookService) {
        this.bookService = bookService;
    }

    @QueryMapping
    public List<Book> books() {
        return bookService.findAll();
    }

    @QueryMapping
    public Book bookById(@Argument Long id) {
        return bookService.findById(id);
    }

    @MutationMapping
    public Book createBook(@Argument @Valid BookInput input) {
        return bookService.create(input);
    }

    @MutationMapping
    public Book updateBook(@Argument Long id, @Argument @Valid BookInput input) {
        return bookService.update(id, input);
    }

    @MutationMapping
    public Boolean deleteBook(@Argument Long id) {
        return bookService.delete(id);
    }
}

Penjelasan alur
Alurnya sederhana:

Client mengirim query atau mutation ke /graphql.

Spring GraphQL mencocokkan field schema ke method @QueryMapping atau @MutationMapping.

Method controller memanggil service.

Service memanggil BookRepository dari Spring Data JPA untuk operasi ke MySQL.

Hasil dikembalikan sebagai response GraphQL.

Contoh mapping ini enak dipakai untuk skripsi atau project backend kecil karena lapisannya jelas: schema -> controller -> service -> repository -> database.

Saran pengembangan
Untuk project yang lebih rapi, kamu bisa tambah:

findByTitleContainingIgnoreCase di repository untuk search.

Exception handler GraphQL dengan @GraphQlExceptionHandler untuk error yang lebih bersih.

Relasi tabel, misalnya Book dan Category, lalu resolver field tambahan dengan @SchemaMapping.

        return bookRepository.findById(id).orElse(null);
    }

    public Book create(BookInput input) {
        Book book = new Book();
        book.setTitle(input.getTitle());
        book.setAuthor(input.getAuthor());
        book.setPrice(input.getPrice());
        return bookRepository.save(book);
    }

    public Book update(Long id, BookInput input) {
        Book book = bookRepository.findById(id)
                .orElseThrow(() -> new RuntimeException("Book tidak ditemukan"));

        book.setTitle(input.getTitle());
        book.setAuthor(input.getAuthor());
        book.setPrice(input.getPrice());

        return bookRepository.save(book);
    }

    public Boolean delete(Long id) {
        if (!bookRepository.existsById(id)) {
            return false;
        }
        bookRepository.deleteById(id);
        return true;
    }
}
6. GraphQL Controller
Repository Spring Data JPA bisa dibuat cukup dengan interface, lalu Spring akan melakukan wiring otomatis; untuk GraphQL, method controller bisa dipetakan ke schema dengan anotasi seperti @QueryMapping dan @MutationMapping.

java
package com.example.graphqlpostgres.controller;

import com.example.graphqlpostgres.entity.Book;
import com.example.graphqlpostgres.input.BookInput;
import com.example.graphqlpostgres.service.BookService;
import jakarta.validation.Valid;
import org.springframework.graphql.data.method.annotation.Argument;
import org.springframework.graphql.data.method.annotation.MutationMapping;
import org.springframework.graphql.data.method.annotation.QueryMapping;
import org.springframework.stereotype.Controller;

import java.util.List;

@Controller
public class BookController {

    private final BookService bookService;

    public BookController(BookService bookService) {
        this.bookService = bookService;
    }

    @QueryMapping
    public List<Book> books() {
        return bookService.findAll();
    }

    @QueryMapping
    public Book bookById(@Argument Long id) {
        return bookService.findById(id);
    }

    @MutationMapping
    public Book createBook(@Argument @Valid BookInput input) {
        return bookService.create(input);
    }

    @MutationMapping
    public Book updateBook(@Argument Long id, @Argument @Valid BookInput input) {
        return bookService.update(id, input);
    }

    @MutationMapping
    public Boolean deleteBook(@Argument Long id) {
        return bookService.delete(id);
    }
}

buatkan query inner join pada graphql
Completed 2 steps
Di GraphQL, konsep “inner join” biasanya tidak ditulis seperti SQL INNER JOIN langsung di query GraphQL umum. Yang dilakukan adalah membuat relasi antar-entity di JPA, lalu di GraphQL kamu ambil data bertingkat, misalnya books { author { ... } }, dan resolver @SchemaMapping akan mengambil data relasinya.

Konsep join
Pada Spring for GraphQL, field di schema bisa dipetakan dengan @SchemaMapping, sehingga field turunan seperti Book.author dapat di-resolve dari object parent Book. @QueryMapping dan @MutationMapping sendiri hanyalah shortcut dari @SchemaMapping untuk type Query dan Mutation.

Jadi kalau di SQL kamu membayangkan:

sql
SELECT b.id, b.title, a.name
FROM books b
INNER JOIN authors a ON b.author_id = a.id;
Di GraphQL bentuknya biasanya seperti ini:

graphql
query {
  books {
    id
    title
    author {
      id
      name
    }
  }
}
GraphQL client meminta data relasi sebagai nested field, bukan menulis sintaks INNER JOIN di query-nya.

import com.example.graphqlpostgres.entity.Author;
import com.example.graphqlpostgres.entity.Book;

import com.example.graphqlpostgres.repository.BookRepository;

import org.springframework.stereotype.Controller;

import java.util.List;

@Controller
public class LibraryController {

    private final BookRepository bookRepository;
    private final AuthorRepository authorRepository;

        this.authorRepository = authorRepository;

    public List<Book> books() {
        return bookRepository.findAll();
    }

    @QueryMapping
    public List<Author> authors() {

    @SchemaMapping

Kalau mau benar-benar pakai JPQL join
Kalau kamu ingin query database-nya eksplisit seperti inner join, buat custom query di repository. Spring Data JPA menyimpan entity dengan repository dan memanfaatkan EntityManager di belakang layar, sehingga custom query juga bisa dipakai saat perlu kontrol lebih spesifik.

package com.example.graphqlpostgres.repository;

import java.util.List;

public interface BookRepository extends JpaRepository<Book, Long> {

    @Query("SELECT b FROM Book b INNER JOIN b.author a")
    List<Book> findAllWithAuthor();

Bentuk SQL tabel
Sebagai gambaran, tabel PostgreSQL bisa seperti ini:

sql
CREATE TABLE authors (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

CREATE TABLE books (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    author_id BIGINT NOT NULL,
    CONSTRAINT fk_book_author
        FOREIGN KEY (author_id)
        REFERENCES authors(id)
);
Dengan foreign key seperti itu, setiap books.author_id harus mengarah ke authors.id, sehingga secara praktik relasi ini mendukung hasil seperti inner join ketika data author diminta bersama book.

Catatan penting
GraphQL standar tidak punya keyword INNER JOIN seperti SQL; pola umumnya adalah:

definisikan relasi di entity JPA,

buat field relasi di schema,

ambil nested field lewat query GraphQL,

dan bila perlu optimalkan pengambilan data lewat @SchemaMapping atau @BatchMapping.
